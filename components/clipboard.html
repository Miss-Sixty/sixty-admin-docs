# Clipboard 文本复制

# 说明

二次封装 clipboard-copy在新窗口打开

# 代码演示

<template>
  <el-input
    v-model="input"
    placeholder="请输入内容"
    type="textarea"
    :rows="3"
  />
  <el-button
    v-clipboard="input"
    v-clipboard:success="successChange"
    v-clipboard:error="errorChange"
    type="primary"
    style="margin-top: 20px"
  >
    点击复制
  </el-button>
</template>
<script setup>
const input = ref("连雨不知春去，一晴方觉夏深。");
const successChange = (res) => ElMessage.success(res);
const errorChange = (res) => ElMessage.error(res);
</script>
